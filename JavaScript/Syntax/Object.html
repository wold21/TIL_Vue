# Object for문 돌리기

let obj = {
    "test1" : 1,
    "test2" : 2,
    "test3" : 3,
};

// 방법 1
for(let key in obj){
     console.log(key);
}

// 방법 2
for(let i=0; i<Object.keys(obj).length; i++){

}
